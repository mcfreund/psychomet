### OPEN FILE FOR AXCPT DATA ###
axcpt_pkl = open('full_dset.pkl', 'rb')
axcpt_dset = cPickle.load(axcpt_pkl)

#Arrays to store (withinSubj - betweenSubj) and p-values respectively
axcpt_comparison = []
axcpt_Pval = []
for parcel in range(0, 400):
    corrMatrix = pd.DataFrame(data=axcpt_dset[parcel][0])
    corrMatrix = corrMatrix.transpose()
    corrMatrix = corrMatrix.corr()
    corrMatrix = np.arctanh(corrMatrix)

    #iterate through correlation matrix and extract within and between subjects
    comparisonVector = []
    for subjRun1 in range(0, 59):
        withinSubj = 0
        btwnSubj = 0
        numBtwnSubj = 58
        for subjRun2 in range(59, 118):
            ### IF SUBJ DATA MISSING ###
            if np.isnan(corrMatrix[subjRun1][subjRun2]):
                numBtwnSubj -= 1
                continue

            ### BETWEEN-RUN WITHIN-SUBJECT CASE ###
            if subjRun2 ==  (subjRun1 + 59):
                withinSubj = corrMatrix[subjRun1][subjRun2]
            ### BETWEEN-RUN BETWEEN-SUBJECT CASE ###
            else:
                btwnSubj += corrMatrix[subjRun1][subjRun2]

        comparisonVector.append(withinSubj - (btwnSubj/numBtwnSubj))

    #append mean of (withingSubj - betweenSubj) and p-values in respective arrays
    axcpt_comparison.append(np.mean(comparisonVector))
    axcpt_Pval.append(stats.ttest_1samp(comparisonVector, 0, nan_policy="omit")[1])
    
#do fdr Benjamini/Hochberg 
axcpt_Pval = mne.stats.fdr_correction(axcpt_Pval)
